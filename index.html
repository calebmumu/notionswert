function displayGreetings(today) { const hrs = today.getHours(); const name = "ADD YOUR NAME HERE"; let greet; if (hrs < 12) greet = 'Good Morning ' + name; else if (hrs >= 12 && hrs <= 17) greet = 'Good Afternoon ' + name; else if (hrs >= 17 && hrs <= 24) greet = 'Good Evening ' + name; // Check if it's currently in light mode or dark mode const isDarkMode = window.matchMedia('(prefers-color-scheme: dark)').matches; // Set the background and text color based on the mode if (isDarkMode) { document.body.style.backgroundColor = '#333'; // Dark background color document.getElementById('greet').style.color = '#fff'; // Light text color } else { document.body.style.backgroundColor = '#fff'; // Light background color document.getElementById('greet').style.color = '#333'; // Dark text color } // Display the greeting document.getElementById('greet').innerHTML = greet; }
